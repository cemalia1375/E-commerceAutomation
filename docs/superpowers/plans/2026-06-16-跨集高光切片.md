# 跨集高光切片 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为 FlowCut 新增「跨集高光切片」：前 3 集合并送 Gemini 自动识别高光起点，从起点逐集向后拼到 50–70s（句子优先收尾），一部剧产出多条 fc_creative 候选并异步 ffmpeg 合成。

**Architecture:** 新增一条 `HIGHLIGHT_PLAN` durable 任务流，由一个规划执行器在服务端串起「合并前3集→Gemini检测起点→逐集前向解→选边界→建 creative→入队 VIDEO_COMPOSE」。所有纯算法（选起点/建时间线/选边界/反算裁切计划）放进无 IO 的 `clip_planner.py`，可完整单测；执行器只做编排与 IO。`video_compose` 执行器扩展一个 `clip_plan_json` 分支做多段裁切+拼接。

**Tech Stack:** Python 3 + asyncio、FastAPI 容器依赖注入、aiomysql（fc_creative）、Google Gemini（`analyze_highlight_video` / `analyze_video`）、ffmpeg（模块内现成 helper）、pytest / pytest-asyncio。

**约定：** 所有命令在 `SimpleClaw/` 目录下运行。测试用 `uv run pytest`。

**对 spec 的一处刻意简化（YAGNI）：** spec 提到「解集结果落 fc_script 缓存」。本计划在 MVP 中**不持久化 fc_script**，改为在单个规划任务内用内存字典对「同一集只解一次」做去重（`_decode_cache`）。跨任务复用缓存属后续优化，避免引入 per-episode ref_video/script 落库耦合。其余 spec 要求全部覆盖。

```python
"""跨集高光切片的纯算法层（无 IO，可完整单测）。

输入是已解好的 segment 字典（来自 gemini_video.analyze_* 输出），
输出是选好的起点、跨集时间线、收尾边界、各集裁切计划。
所有 IO（下载/Gemini/ffmpeg/落库/入队）由 executor 负责，本模块不碰。
"""
from __future__ import annotations

from dataclasses import dataclass

START_SEARCH_EPISODES = 3
WINDOW = (50.0, 70.0)
IDEAL = 60.0
DEFAULT_CANDIDATES = 3
MAX_FORWARD_EPISODES = 6
DEDUP_GAP_S = 2.0
SENTENCE_PUNCT = "。！？…!?."
_PRIMARY = {"primary_hook", "secondary_hook"}


@dataclass(frozen=True)
class EpisodeRef:
    asset_id: int
    episode_no: int
    oss_key: str
    duration: float  # 秒


@dataclass(frozen=True)
class StartCandidate:
    episode_no: int
    local_start: float
    global_start: float
    hook_strength: float
    reason: str


def _locate(global_t: float, offsets: list[tuple[int, float]],
            durations: dict[int, float]) -> tuple[int, float] | None:
    """把前3集拼接视频里的全局秒数映射回 (episode_no, 集内秒)。"""
    for ep_no, off in offsets:
        dur = durations.get(ep_no, 0.0)
        if off <= global_t < off + dur:
            return ep_no, global_t - off
    # 落在最后一集末尾的边界容差
    if offsets:
        ep_no, off = offsets[-1]
        return ep_no, max(0.0, global_t - off)
    return None


def select_start_candidates(
    segments: list[dict],
    offsets: list[tuple[int, float]],
    durations: dict[int, float],
    top_n: int,
) -> list[StartCandidate]:
    """从前3集合并解出的 segments 里选 top-N 高光起点。

    排序：primary/secondary_hook 优先 → hook_strength 高 → context_dependency 低
    → continuity_risk 低。去重：全局起点相距 < DEDUP_GAP_S 的只留排名更高者。
    """
    ranked = sorted(
        (s for s in segments if str(s.get("candidate_use")) in _PRIMARY),
        key=lambda s: (
            0 if str(s.get("candidate_use")) == "primary_hook" else 1,
            -float(s.get("hook_strength") or 0.0),
            float(s.get("context_dependency") or 0.0),
            float(s.get("continuity_risk") or 0.0),
        ),
    )
    out: list[StartCandidate] = []
    for s in ranked:
        g = float(s.get("start_time") or 0.0)
        if any(abs(g - c.global_start) < DEDUP_GAP_S for c in out):
            continue
        loc = _locate(g, offsets, durations)
        if loc is None:
            continue
        ep_no, local = loc
        out.append(StartCandidate(
            episode_no=ep_no,
            local_start=local,
            global_start=g,
            hook_strength=float(s.get("hook_strength") or 0.0),
            reason=str(s.get("reason") or ""),
        ))
        if len(out) >= top_n:
            break
    return out
```

@dataclass(frozen=True)
class TimedSegment:
    episode_no: int
    seg_start: float  # 集内秒
    seg_end: float    # 集内秒
    cum_start: float  # 从切片起点(=0)起的全局累计
    cum_end: float
    copy: str


def build_timeline(
    start_local: float,
    ordered_eps: list[tuple["EpisodeRef", list[dict]]],
) -> list[TimedSegment]:
    """从起点集的 start_local 起，按集顺序拼成带累计时间的 segment 列表。

    第一个元素必须是起点集：只保留 seg_end > start_local 的段，首段裁到 start_local。
    后续集全部纳入。cum 从 0 连续累加。
    """
    out: list[TimedSegment] = []
    cum = 0.0
    for i, (ep, segs) in enumerate(ordered_eps):
        ordered = sorted(segs, key=lambda s: float(s.get("start_time") or 0.0))
        for s in ordered:
            seg_start = float(s.get("start_time") or 0.0)
            seg_end = float(s.get("end_time") or 0.0)
            if i == 0:
                if seg_end <= start_local:
                    continue
                seg_start = max(seg_start, start_local)
            if seg_end <= seg_start:
                continue
            dur = seg_end - seg_start
            out.append(TimedSegment(
                episode_no=ep.episode_no,
                seg_start=seg_start,
                seg_end=seg_end,
                cum_start=cum,
                cum_end=cum + dur,
                copy=str(s.get("copy") or ""),
            ))
            cum += dur
    return out
```

@dataclass(frozen=True)
class EndBoundary:
    cum_time: float
    episode_no: int
    local_end: float
    boundary_type: str  # 'sentence' | 'shot' | 'hard_cut'


def _is_sentence_safe(copy: str) -> bool:
    """该段结尾是否「说话安全」：无台词，或台词以句末标点结束。"""
    text = copy.strip()
    if not text:
        return True
    return text[-1] in SENTENCE_PUNCT


def pick_end_boundary(
    timeline: list[TimedSegment],
    window: tuple[float, float] = WINDOW,
    ideal: float = IDEAL,
) -> EndBoundary:
    """在 [window] 内选收尾边界：句子安全优先 → 分镜切点其次 → hard_cut 兜底。"""
    lo, hi = window
    in_window = [ts for ts in timeline if lo <= ts.cum_end <= hi]

    safe = [ts for ts in in_window if _is_sentence_safe(ts.copy)]
    if safe:
        best = min(safe, key=lambda ts: abs(ts.cum_end - ideal))  # 句末安全边界里取最接近 ideal 的
        return EndBoundary(best.cum_end, best.episode_no, best.seg_end, "sentence")

    if in_window:
        best = min(in_window, key=lambda ts: abs(ts.cum_end - ideal))
        return EndBoundary(best.cum_end, best.episode_no, best.seg_end, "shot")

    # 窗口内无任何 segment 边界（极端超长镜头）→ 在 ideal 处硬切
    for ts in timeline:
        if ts.cum_start <= ideal < ts.cum_end:
            local = ts.seg_start + (ideal - ts.cum_start)
            return EndBoundary(ideal, ts.episode_no, local, "hard_cut")
    # ideal 超出整条时间线（剧太短）→ 用最后一段末尾
    last = timeline[-1]
    return EndBoundary(last.cum_end, last.episode_no, last.seg_end, "hard_cut")
```

@dataclass(frozen=True)
class ClipEntry:
    asset_id: int
    episode_no: int
    cut_start: float
    cut_end: float
    oss_key: str


@dataclass(frozen=True)
class ClipPlan:
    entries: tuple[ClipEntry, ...]
    total_duration: float
    boundary_type: str
    start_episode_no: int
    start_local: float


def build_clip_plan(
    start: StartCandidate,
    end: EndBoundary,
    episodes: list["EpisodeRef"],
) -> ClipPlan:
    """把 (起点, 收尾边界) 反算成按 episode_no 顺序的逐集裁切计划。

    episodes 必须是从起点集到收尾集（含）的有序 EpisodeRef 列表。
    """
    entries: list[ClipEntry] = []
    for ep in episodes:
        if ep.episode_no < start.episode_no or ep.episode_no > end.episode_no:
            continue
        is_start = ep.episode_no == start.episode_no
        is_end = ep.episode_no == end.episode_no
        cut_start = start.local_start if is_start else 0.0
        cut_end = end.local_end if is_end else ep.duration
        if cut_end <= cut_start:
            continue
        entries.append(ClipEntry(
            asset_id=ep.asset_id, episode_no=ep.episode_no,
            cut_start=cut_start, cut_end=cut_end, oss_key=ep.oss_key,
        ))
    return ClipPlan(
        entries=tuple(entries),
        total_duration=end.cum_time,
        boundary_type=end.boundary_type,
        start_episode_no=start.episode_no,
        start_local=start.local_start,
    )

def make_highlight_plan_executor(
    *,
    runtime,
    highlight_asset_repo,
    creative_repo,
    oss_client,
) -> Callable[[TaskEnvelope], Awaitable[TaskExecutionResult]]:
    """跨集高光切片规划：合并前3集→Gemini选起点→逐集前向解→选边界→建 creative→入队 VIDEO_COMPOSE。"""

    async def _decode_episode(asset: dict, tmp_dir: str, cache: dict, *, highlight: bool):
        """解一集，返回 (segments, local_path, duration)；同一 asset 只解一次。"""
        asset_id = int(asset["id"])
        if asset_id in cache:
            return cache[asset_id]
        loop = asyncio.get_running_loop()
        path = os.path.join(tmp_dir, f"ep_{asset_id}.mp4")
        await loop.run_in_executor(
            None, oss_client.download,
            str(asset.get("oss_key") or asset.get("oss_url") or ""), path,
        )
        duration = await loop.run_in_executor(None, _probe_duration_seconds, path)
        if highlight:
            result = await analyze_highlight_video(path, continuation_type="original")
            segments = result.get("segments", [])
        else:
            segments = await analyze_video(path, preserve_extra=True)
        cache[asset_id] = (segments, path, float(duration))
        return cache[asset_id]

    async def execute(task: TaskEnvelope) -> TaskExecutionResult:
        payload = task.payload
        tenant_key = str(payload.get("tenant_key") or task.tenant_key or "flowcut")
        drama_name = str(payload.get("drama_name") or "").strip()
        session_key = str(payload.get("session_key") or "highlight_plan")
        num_candidates = int(payload.get("num_candidates") or DEFAULT_CANDIDATES)
        batch_id = str(payload.get("batch_id") or uuid.uuid4().hex)

        if not drama_name:
            return TaskExecutionResult.failed(error="highlight_plan: drama_name 为空")

        rows = await highlight_asset_repo.list_by_tenant(
            tenant_key, asset_type="episode_source", drama_name=drama_name, limit=500,
        )
        episodes = sorted(rows, key=lambda r: int(r.get("episode_no") or 0))
        if not episodes:
            return TaskExecutionResult.failed(error=f"没有在原片库找到「{drama_name}」")

        tmp_dir = tempfile.mkdtemp(prefix=f"flowcut_highlight_plan_{batch_id}_")
        decode_cache: dict[int, tuple] = {}
        try:
            # —— Phase 1：合并前3集 → 一次 Gemini 高光检测 ——
            head = episodes[:START_SEARCH_EPISODES]
            loop = asyncio.get_running_loop()
            norm_paths: list[str] = []
            offsets: list[tuple[int, float]] = []
            durations: dict[int, float] = {}
            cum = 0.0
            for asset in head:
                ep_path = os.path.join(tmp_dir, f"head_{asset['id']}.mp4")
                await loop.run_in_executor(
                    None, oss_client.download,
                    str(asset.get("oss_key") or asset.get("oss_url") or ""), ep_path,
                )
                norm_path = os.path.join(tmp_dir, f"head_{asset['id']}_norm.mp4")
                _ffmpeg_normalize_clip(ep_path, norm_path)
                dur = await loop.run_in_executor(None, _probe_duration_seconds, norm_path)
                norm_paths.append(norm_path)
                offsets.append((int(asset["episode_no"] or 0), cum))
                durations[int(asset["episode_no"] or 0)] = float(dur)
                cum += float(dur)

            merged_path = os.path.join(tmp_dir, "head_merged.mp4")
            concat_list = os.path.join(tmp_dir, "head_concat.txt")
            _write_concat_list(concat_list, norm_paths)
            _ffmpeg_concat(concat_list, merged_path)

            merged_result = await analyze_highlight_video(merged_path, continuation_type="original")
            head_segments = merged_result.get("segments", [])
            # 把前3集 segment 按集存进 decode_cache（按 offset 切回各集本地时间）
            ep_no_to_asset = {int(a["episode_no"] or 0): a for a in head}
            for ep_no, off in offsets:
                local_segs = [
                    {**s, "start_time": float(s["start_time"]) - off,
                     "end_time": float(s["end_time"]) - off}
                    for s in head_segments
                    if off <= float(s.get("start_time") or 0.0) < off + durations[ep_no]
                ]
                a = ep_no_to_asset[ep_no]
                decode_cache[int(a["id"])] = (local_segs, "", durations[ep_no])

            candidates = select_start_candidates(
                head_segments, offsets, durations, top_n=num_candidates,
            )
            if not candidates:
                return TaskExecutionResult.failed(
                    error=f"「{drama_name}」前{len(head)}集未找到可用高光起点")

            # —— Phase 2：每个候选逐集前向解 → 时间线 → 边界 → clip_plan → creative ——
            ep_index = {int(a["episode_no"] or 0): a for a in episodes}
            created: list[dict] = []
            for cand in candidates:
                ordered_eps: list[tuple] = []
                ep_refs: list[EpisodeRef] = []
                forward_count = 0
                total = 0.0
                ep_no = cand.episode_no
                while ep_no in ep_index and forward_count < MAX_FORWARD_EPISODES:
                    asset = ep_index[ep_no]
                    is_head = int(asset["id"]) in decode_cache and decode_cache[int(asset["id"])][1] == ""
                    segs, _path, dur = await _decode_episode(
                        asset, tmp_dir, decode_cache, highlight=False,
                    ) if not is_head else decode_cache[int(asset["id"])]
                    ref = EpisodeRef(
                        asset_id=int(asset["id"]), episode_no=ep_no,
                        oss_key=str(asset.get("oss_key") or asset.get("oss_url") or ""),
                        duration=float(dur),
                    )
                    ep_refs.append(ref)
                    ordered_eps.append((ref, segs))
                    # 累计时长估计：起点集从 local_start 起
                    if ep_no == cand.episode_no:
                        total += max(0.0, float(dur) - cand.local_start)
                    else:
                        total += float(dur)
                    forward_count += 1
                    if total >= WINDOW[1]:
                        break
                    ep_no += 1

                timeline = build_timeline(cand.local_start, ordered_eps)
                if not timeline or timeline[-1].cum_end < WINDOW[0]:
                    logger.info("highlight_plan: 候选起点 ep=%d 向后不足 %ss，跳过",
                                cand.episode_no, WINDOW[0])
                    continue
                end = pick_end_boundary(timeline)
                plan = build_clip_plan(cand, end, ep_refs)
                if not plan.entries:
                    continue

                clip_plan_dict = {
                    "drama_name": drama_name,
                    "boundary_type": plan.boundary_type,
                    "total_duration": plan.total_duration,
                    "start_episode_no": plan.start_episode_no,
                    "start_local": plan.start_local,
                    "entries": [
                        {"asset_id": e.asset_id, "episode_no": e.episode_no,
                         "cut_start": e.cut_start, "cut_end": e.cut_end,
                         "oss_key": e.oss_key}
                        for e in plan.entries
                    ],
                }
                reason_dict = {
                    "hook_strength": cand.hook_strength,
                    "reason": cand.reason,
                    "boundary_type": plan.boundary_type,
                }
                creative = await creative_repo.create_cross_episode_job(
                    tenant_key=tenant_key,
                    session_key=session_key,
                    script_id=None,
                    batch_id=batch_id,
                    source_asset_id=ep_index[cand.episode_no]["id"],
                    clip_plan_json=json.dumps(clip_plan_dict, ensure_ascii=False),
                    highlight_start=cand.local_start,
                    highlight_reason_json=json.dumps(reason_dict, ensure_ascii=False),
                )
                creative_id = int(creative["id"])
                envelope = TaskEnvelope(
                    task_type="highlight_compose",
                    payload={"creative_id": creative_id},
                    stream=FlowcutTaskStream.VIDEO_COMPOSE,
                    tenant_key=tenant_key,
                    session_key=session_key or None,
                    scope_key=f"cross_highlight_compose:{creative_id}",
                )
                await runtime.submit_task(envelope)
                created.append({"creative_id": creative_id,
                                "start_episode_no": cand.episode_no,
                                "boundary_type": plan.boundary_type})

            if not created:
                return TaskExecutionResult.failed(
                    error=f"「{drama_name}」所有候选都无法凑到 {WINDOW[0]}s")

            return TaskExecutionResult.succeeded(
                summary=f"cross-episode highlight plan drama={drama_name} created={len(created)}",
                details={"drama_name": drama_name, "batch_id": batch_id,
                         "created": created},
            )
        except Exception as exc:
            return TaskExecutionResult.failed(error=f"{type(exc).__name__}: {exc}")
        finally:
            shutil.rmtree(tmp_dir, ignore_errors=True)

    return execute
```

"""create_cross_episode_highlights 工具：跨集高光切片规划任务。"""
from __future__ import annotations

import json
import uuid
from typing import TYPE_CHECKING, Any

from simpleclaw.runtime.task_protocol import TaskEnvelope
from simpleclaw.tools.base import Tool, ToolResult

from Flowcut.runtime.streams import FlowcutTaskStream

if TYPE_CHECKING:
    from simpleclaw.runtime.services import RuntimeServices


class CreateCrossEpisodeHighlightsTool(Tool):
    name = "create_cross_episode_highlights"
    description = (
        "按 AI 漫剧名称，从前 3 集自动识别高光起点，跨集向后拼接约 1 分钟的连续切片，"
        "一次产出多条候选成片。用户要求「跨集高光」「从前几集抽一分钟」「连续切片」时使用。"
    )
    parameters = {
        "type": "object",
        "properties": {
            "drama_name": {"type": "string", "description": "AI 漫剧名称"},
            "num_candidates": {
                "type": "integer",
                "description": "产出几条候选切片，默认 3",
                "default": 3,
            },
        },
        "required": ["drama_name"],
    }
    execution_mode = "durable"
    needs_followup = True
    durable_action = "queued"
    tool_category = "background_write"
    read_only = False

    def __init__(self, *, runtime: "RuntimeServices") -> None:
        del runtime  # 入队走 RuntimeServices（基类注入），此处不直接持有
        self._tenant_key = "flowcut"
        self._session_key = ""

    def set_context(self, *, tenant_key: str = "", session_key: str = "", **_: object) -> None:
        if tenant_key:
            self._tenant_key = tenant_key
        self._session_key = session_key or self._session_key

    async def prepare_task(
        self, drama_name: str, num_candidates: int = 3, **_: Any,
    ) -> TaskEnvelope | ToolResult:
        drama = (drama_name or "").strip()
        if not drama:
            return ToolResult(
                content=json.dumps({"ok": False, "error": "drama_name 不能为空"},
                                   ensure_ascii=False),
                ok=False,
            )
        capped = max(1, min(int(num_candidates or 3), 10))
        batch_id = uuid.uuid4().hex
        return TaskEnvelope(
            task_type="highlight_plan",
            payload={
                "drama_name": drama,
                "num_candidates": capped,
                "tenant_key": self._tenant_key,
                "session_key": self._session_key,
                "batch_id": batch_id,
            },
            stream=FlowcutTaskStream.HIGHLIGHT_PLAN,
            tenant_key=self._tenant_key,
            session_key=self._session_key or None,
            scope_key=f"highlight_plan:{drama}:{batch_id}",
        )

    def durable_result(self, task: TaskEnvelope, queue_id: str) -> ToolResult:
        return ToolResult(
            content=json.dumps(
                {
                    "ok": True,
                    "action": self.durable_action,
                    "task_id": task.task_id,
                    "task_type": task.task_type,
                    "queue_id": queue_id,
                    "data": {
                        "drama_name": task.payload.get("drama_name"),
                        "num_candidates": task.payload.get("num_candidates"),
                    },
                    "navigate": {"route": "/creative", "mode": "push"},
                    "ui_hint": {"render_as": "text", "title": "已开始跨集高光切片"},
                },
                ensure_ascii=False,
            )
        )
```
